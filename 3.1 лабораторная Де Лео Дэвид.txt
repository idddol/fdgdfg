void mergesort(int* a, int l, int r) {
	int middle, startl, startr, i;
	int mas[1000] = { 0 };
	middle = (l + r) / 2; //вычисление среднего элемента
	startl = l;//начало левой части
	startr = middle + 1;//начало правой части
	cout << endl;
	for (i = l; i < r + 1; i++) {
		if ((startl < middle) && (startr < r + 1)) {
			if (a[startl] < a[startr]) {
				mas[i] = a[startl];
				startl++;
			}
			else {
				mas[i] = a[startr];
				startr++;
			}
		}
		else {
			if (startl < middle + 1) {
				mas[i] = a[startl];
				startl++;
			}
			else {
				mas[i] = a[startr];
				startr++;
			}
		}
	}
	for (i = l; i < r + 1; i++) a[i] = mas[i];//возращение результата
}

void Task3(int* a, int l, int r) {
	if (r - l < 1) {
		Task3(a, l, (l + r) / 2);//сортировка левой части
		Task3(a, (l + r) / 2 + 1, r);//сортировка правой части
		mergesort(a, l, r); // слияние
	}
	else {
		if (l == r) return;
		else {
			if (a[r] < a[l]) swap(a[r], a[l]);
		}
	}
}
